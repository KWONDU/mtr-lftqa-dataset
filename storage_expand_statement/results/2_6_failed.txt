def expand_statement_pattern(df):
    statement_pattern = "The top scorer of the {series} for the {caption} is {leading_scorer}."
    expanded_statement_list = []

    for index, row in df.iterrows():

        caption = row['Caption']
        leading_scorer = df[df['Series'] == series]['Leading_Scorer'].iloc[0]
        expanded_statement = statement_pattern.format(series=series, caption=caption, leading_scorer=leading_scorer)
