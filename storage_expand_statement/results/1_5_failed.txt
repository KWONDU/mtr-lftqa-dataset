import pandas as pd

def expand_statement_pattern(df):
    statement_pattern = "Real Madrid CF's total number of assists for the 2009-10 season with the position as {position} and the league {comparative} {league_threshold} is {assist_count}."
    expanded_statement_list = []

    for index, row in df.iterrows():
        position = row['Position_']
        league_threshold = 7.0
        assist_count = df[(df['Position_'] == position) & (df['League'] > league_threshold)]['Total'].sum()
        comparative = "greater than" if assist_count > 1 else "less than or equal to"
        expanded_statement = statement_pattern.format(position=position, comparative=comparative, league_threshold=league_threshold, assist_count=assist_count)
        expanded_statement_list.append(expanded_statement)

    return statement_pattern, expanded_statement_list
