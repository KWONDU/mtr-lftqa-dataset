import pandas as pd

def expand_statement_pattern(df):
    statement_pattern = "The driver who placed best for {team_name} in the 1998 Pepsi 400 was {driver_name}."
    expanded_statement_list = []

    for index, row in df.iterrows():
        team_name = row['Team']
        driver_name = df.loc[df['Team'] == team_name, 'Driver'].iloc[0]
        expanded_statement = statement_pattern.format(team_name=team_name, driver_name=driver_name)
        expanded_statement_list.append(expanded_statement)

    return statement_pattern, expanded_statement_list
