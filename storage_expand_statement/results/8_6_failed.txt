import pandas as pd

def expand_statement_pattern(df):
    statement_pattern = "In {date}, on the {team_name} football team, the Brazilian goalscorer with a score of {score}, a competition of {competition}, and a date of {date} is {player_name}."
    expanded_statement_list = []

    for index, row in df.iterrows():
        date = row['Date_']
        team_name = row['Brazil_scorers'].split('(')[0].strip()
        score = row['Score']
        competition = row['Competition']
        player_name = row['Brazil_scorers'].split('(')[1].split(')')[0]
        expanded_statement = statement_pattern.format(date=date, team_name=team_name, score=score, competition=competition, player_name=player_name)
        expanded_statement_list.append(expanded_statement)

    return statement_pattern, expanded_statement_list
