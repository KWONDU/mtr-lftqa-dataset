import pandas as pd

def expand_statement_pattern(df):
    statement_pattern = "The Brazil national team's result under {group} in the {year} {competition} in Brazilian soccer on {date} was a {result}."
    expanded_statement_list = []

    for index, row in df.iterrows():
        group = row['Result_']
        year = row['Date_'].split(',')[1].strip()
        competition = row['Competition']
        date = "21 March 1954"  # Specific date mentioned in the statement
        result = "win"  # Specific result mentioned in the statement

        expanded_statement = statement_pattern.format(group=group, year=year, competition=competition, date=date, result=result)
        expanded_statement_list.append(expanded_statement)

    return statement_pattern, expanded_statement_list
