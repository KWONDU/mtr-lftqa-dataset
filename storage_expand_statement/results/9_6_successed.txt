import pandas as pd

def expand_statement_pattern(df):
    statement_pattern = "No grid has a {team_name} team and laps greater than {laps} for the 2005 Time Warner Cable Road Runner 225."
    expanded_statement_list = []

    for index, row in df.iterrows():
        team_name = row['Team']
        laps = row['Laps']
        filtered_df = df[(df['Team'] == team_name) & (df['Laps'] > laps)]
        if filtered_df.empty:
            expanded_statement = statement_pattern.format(team_name=team_name, laps=laps)
            expanded_statement_list.append(expanded_statement)

    return statement_pattern, expanded_statement_list
